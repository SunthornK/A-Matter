# Database & Schema Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.
> **Read CONTEXT.md first** — it contains the full schema spec, game rules, and architectural decisions.

**Goal:** Set up PostgreSQL + Prisma in a new `packages/db` workspace package, define the full schema (users, rooms, games, game_players, moves), generate the Prisma client, write a seed script, and implement a `replayGame()` function that reconstructs full game state from the moves log.

**Architecture:** `packages/db` exports the Prisma client and all DB-layer utilities. All other packages (`server`, etc.) import from `@a-matter/db`. The `moves` table is the single source of truth — `board_state`, `tile_bag`, `rack`, and `score` in `games`/`game_players` are derived caches written atomically per move.

**Tech Stack:** PostgreSQL 15+, Prisma 5, TypeScript 5, Vitest (integration tests against a real test DB)

```typescript
// packages/db/src/replayGame.ts
import { prisma } from './client'
import type { BoardState, TileBagEntry, BoardCell, MovePlacement } from './types'
import type { MoveAction } from '@prisma/client'

export interface ReplayResult {
  boardState: BoardState
  tileBag: TileBagEntry[]
  /** rack per player_id */
  racks: Map<string, TileBagEntry[]>
  /** score per player_id */
  scores: Map<string, number>
  turnNumber: number
}

/**
 * Reconstructs full game state by replaying all moves from the moves log.
 * The moves table is the single source of truth.
 * Use this for integrity checks, replay viewer, and disaster recovery.
 */
export async function replayGame(gameId: string): Promise<ReplayResult> {
  const game = await prisma.game.findUnique({
    where: { id: gameId },
    include: {
      players: { orderBy: { seat: 'asc' } },
      moves: { orderBy: { turnNumber: 'asc' } },
    },
  })

  if (!game) throw new Error(`Game not found: ${gameId}`)

  // Initialize empty board
  const board: (BoardCell | null)[][] = Array.from({ length: 15 }, () =>
    Array.from({ length: 15 }, () => null),
  )

  // Initialize full tile bag from stored initial state
  // The initial tileBag in games table has all 100 tiles
  const initialBag = game.tileBag as unknown as TileBagEntry[]
  const bagMap = new Map<string, TileBagEntry>(initialBag.map((t) => [t.tile_id, t]))

  // Track per-player state
  const racks = new Map<string, TileBagEntry[]>()
  const scores = new Map<string, number>()
  for (const player of game.players) {
    racks.set(player.id, [])
    scores.set(player.id, 0)
  }

  let turnNumber = 1

  for (const move of game.moves) {
    const action = move.action as MoveAction

    if (action === 'place') {
      const placements = move.placements as unknown as MovePlacement[]
      if (placements) {
        for (const placement of placements) {
          const tile = bagMap.get(placement.tile_id)
          if (tile) {
            const cell: BoardCell = {
              ...tile,
              display_value: placement.dual_choice ?? placement.blank_designation ?? tile.value,
              is_blank: tile.type === 'blank',
              blank_designation: placement.blank_designation,
              dual_choice: placement.dual_choice,
            }
            board[placement.row]![placement.col] = cell
            bagMap.delete(placement.tile_id)

            // Remove from player rack
            const rack = racks.get(move.playerId) ?? []
            racks.set(move.playerId, rack.filter((t) => t.tile_id !== placement.tile_id))
          }
        }
        // Update score
        scores.set(move.playerId, (scores.get(move.playerId) ?? 0) + move.scoreEarned)
      }
    } else if (action === 'exchange') {
      const indices = move.exchangedIndices as unknown as number[]
      if (indices) {
        const rack = racks.get(move.playerId) ?? []
        // Tiles at those indices go back to bag, then player draws new ones
        // For replay purposes: just note that exchange happened
        // (exact new tiles aren't tracked in moves — they're random draws)
        const remaining = rack.filter((_, i) => !indices.includes(i))
        racks.set(move.playerId, remaining)
      }
    }
    // 'pass' — nothing changes on board or bag

    turnNumber = move.turnNumber + 1
  }

  const tileBag = Array.from(bagMap.values())

  return {
    boardState: { cells: board },
    tileBag,
    racks,
    scores,
    turnNumber,
  }
}
```

```typescript
  // Create a completed ranked game
  const room = await prisma.room.create({
    data: {
      creatorId: alice.id,
      type: 'ranked',
      status: 'closed',
      expiresAt: new Date(Date.now() + 1000),
    },
  })

  const emptyBoard = { cells: Array.from({ length: 15 }, () => Array(15).fill(null)) }

  const game = await prisma.game.create({
    data: {
      roomId: room.id,
      mode: 'ranked',
      status: 'finished',
      boardState: emptyBoard,
      tileBag: [],
      turnNumber: 8,
      endReason: 'stalemate',
      finishedAt: new Date(),
    },
  })

  const player1 = await prisma.gamePlayer.create({
    data: {
      gameId: game.id,
      userId: alice.id,
      seat: 1,
      score: 72,
      rack: [],
      timeRemainingMs: 900000,
    },
  })

  const player2 = await prisma.gamePlayer.create({
    data: {
      gameId: game.id,
      userId: bob.id,
      seat: 2,
      score: 55,
      rack: [],
      timeRemainingMs: 1100000,
    },
  })

  // Add a few pass moves to simulate stalemate
  for (let turn = 1; turn <= 6; turn++) {
    const playerId = turn % 2 === 1 ? player1.id : player2.id
    await prisma.move.create({
      data: {
        gameId: game.id,
        playerId,
        turnNumber: turn,
        action: 'pass',
        scoreEarned: 0,
        timeSpentMs: 3000,
      },
    })
  }

  console.log(`Created game ${game.id} (stalemate, alice wins 72-55)`)
  console.log('Seeding complete.')
}

main()
  .catch((e) => {
    console.error(e)
    process.exit(1)
  })
  .finally(() => prisma.$disconnect())
```
